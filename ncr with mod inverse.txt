const int N = 1e5+10, M = 1050, inf = 1e6, mod = 1e9+7;
int add ( int a , int b)
{
    return (a + b) % mod;
}
int mul ( int a , int b)
{
    return 1LL * a * b % mod;
}
int fp( int b , int p)
{
    if(!p)
        return 1;
    int temp = fp(b,p/2);
    temp = mul(temp,temp);
    if(p&1)
        temp = mul(temp,b);
    return temp;
}
int fact[N], inv[N];
int nCr( int n , int r)
{
    return mul(fact[n],mul(inv[n-r],inv[r]));
}
void testCase() {
    ll n , k ;
    cin >> n >> k;
    fact[0] = inv[0] = 1;
    for ( int i = 1 ; i <= n ; i++)
    {
        fact[i] = mul(i,fact[i-1]);
        inv[i] = fp(fact[i],mod-2);
    }
    int ans = fp(2,n);
    if( n <= k)
        return cout << ans << '\n',void();
    for ( int i = n ; i > k; i--)
        ans = add(ans,mod-nCr(n,i));
    cout << ans << '\n';
}
int32_t main()
{
    Tsetso
    int t = 1;
//    cin >> t;
    while(t--)
        testCase();
}
